### LSP hook 加固应用

通过 hook Application 的 attch 函数，获得上下文，从上下文中获取 classloader

```Java
XposedHelpers.findAndHookMethod(Application.class, "attach", Context.class, new XC_MethodHook() {  
    @Override  
    protected void afterHookedMethod(MethodHookParam param) throws Throwable {  
        Context context = (Context) param.args[0];  
        ClassLoader cl = context.getClassLoader();  
        try {  
            Class<?> hookClass = cl.loadClass("com.xxx.xxx");  
            XposedHelpers.findAndHookMethod(hookClass, "func", new XC_MethodHook() {  
                @Override  
                protected void afterHookedMethod(MethodHookParam param) {  
                    Log.d("duck", "hook成功");  
                }            
            });            
            Log.d("duck", "class found and hooked");  
        } catch (ClassNotFoundException e) {  
            Log.e("duck", "class not found yet");  
        }    
    }
});

XposedHelpers.findAndHookMethod(Application.class, "attach", Context.class, new XC_MethodHook() {  
    @Override  
    protected void afterHookedMethod(MethodHookParam param) throws Throwable {  
        Context context = (Context) param.args[0];  
        ClassLoader cl = context.getClassLoader();  
        XposedHelpers.findAndHookMethod(ClassLoader.class, "loadClass", String.class, new XC_MethodHook() {  
            @Override  
            protected void afterHookedMethod(MethodHookParam param) throws Throwable {  
                String name = (String) param.args[0];  
                if (name.equals("com.xxx.xxx")) {  
                    Class<?> hookClass = (Class<?>) param.getResult();  
                    XposedHelpers.findAndHookMethod(hookClass, "func",  
                            new XC_MethodHook() {  
                                @Override  
                                protected void afterHookedMethod(MethodHookParam param) {  
                                    Log.d("duck", "hook成功");  
                                }                            
                            });                    
                            Log.d("duck", "real class hooked");  
								
                }     
            }        
        });    
	}
});

```
